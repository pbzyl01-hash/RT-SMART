// GET PIN ADMIN FROM DATABASE
function getAdminPin() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Pengaturan");
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === "ADMIN_PIN") {
      return data[i][1].toString();
    }
  }
  return "1234";
}

// UPDATE PIN ADMIN
function updateAdminPin(oldPin, newPin) {
  var currentPin = getAdminPin();
  if (currentPin !== oldPin.toString()) {
    return { success: false, message: "PIN lama yang Anda masukkan salah!" };
  }
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Pengaturan");
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === "ADMIN_PIN") {
      sheet.getRange(i + 1, 2).setValue(newPin.toString());
      return { success: true };
    }
  }
  return { success: false, message: "Terjadi kesalahan sistem pengaturan PIN." };
}

// CRUD CITIZENS WITH EDIT CAPABILITY
function getCitizens() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Warga");
  var data = sheet.getDataRange().getValues();
  data.shift();
  return data.map(function(r) {
    return { nama: r[0], nik: r[1], alamat: r[2], hp: r[3], status: r[4] };
  });
}

function updateOrAddCitizen(originalNik, citizen) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Warga");
  var data = sheet.getDataRange().getValues();
  
  if (originalNik) {
    // Mode Edit: cari baris yang lama berdasarkan original NIK
    for (var i = 1; i < data.length; i++) {
      if (data[i][1].toString() === originalNik.toString()) {
        sheet.getRange(i + 1, 1, 1, 5).setValues([[citizen.nama, citizen.nik, citizen.alamat, citizen.hp, citizen.status]]);
        return { success: true, mode: "edit" };
      }
    }
  }
  // Mode Tambah Baru
  sheet.appendRow([citizen.nama, citizen.nik, citizen.alamat, citizen.hp, citizen.status]);
  return { success: true, mode: "add" };
}

// CRUD FINANCIAL LEDGER
function getFinances() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Keuangan");
  var data = sheet.getDataRange().getValues();
  data.shift();
  return data.map(function(r) {
    return { tanggal: r[0], keterangan: r[1], tipe: r[2], jumlah: parseFloat(r[3]) };
  });
}

function updateOrAddFinance(index, item) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Keuangan");
  if (index !== -1 && index !== "-1") {
    var row = parseInt(index) + 2; // +1 headers, +1 base row index
    sheet.getRange(row, 1, 1, 4).setValues([[item.tanggal, item.keterangan, item.tipe, item.jumlah]]);
    return { success: true, mode: "edit" };
  }
  sheet.appendRow([item.tanggal, item.keterangan, item.tipe, item.jumlah]);
  return { success: true, mode: "add" };
}

}

function updateOrAddAnnouncement(index, ann) {

    sheet.getRange(row, 1, 1, 4).setValues([[ann.tanggal, ann.judul, ann.konten, ann.lampiranUrl]]);

  }
  sheet.appendRow([ann.tanggal, ann.judul, ann.konten, ann.lampiranUrl]);
  return { success: true, mode: "add" };
}

Cara Penggunaan:

  1. Buka spreadsheet Google Sheets Anda.
  2. Klik menu Extensions -> Apps Script.
  3. Ganti semua kode di file `Code.gs` dengan kode dari tab Code.gs di samping.
  4. Klik tanda tambah -> Tambah file HTML bernama Index (tanpa .html).
  5. Ganti semua kode di file `Index.html` dengan seluruh kode dari file ini.
  6. Klik Save, lalu buatlah Deployment Web App baru!
